vector<int> findDiagonalOrder(vector<vector<int>>& mat) {
		vector<int> r;
		for (int i = 1; i < mat.size() + mat[0].size(); i++) {
				for (int k = 0; k < i; k++) {
						if (!(i-1) || (i-1) % 2) {
							if (k < mat.size() && (i-1 - k) < mat[0].size()) 
							r.push_back(mat[k][i-1 - k]);
						}
						else {
							if (i-1-k < mat.size() && k < mat[0].size())
							r.push_back(mat[i-1-k][k]);
						}
				}
		}
		return r;
	}
