void BFS_shortestPath(Graph G,int first) {
	//åˆå§‹åŒ–éƒ¨åˆ†
	std::queue<int> q;
	visited[first] = 1;
	distance[first] = 0;
	q.push(first);
	while (!q.empty()) {
		int current = q.front();
		q.pop();
		for (int w = firstNeighbor(G, current); w >= 0; w = nextNeighbor(G, current, w)) {
			if (!visited[w]) {
				q.push(w);
				visited[w] = 1;
				distance[w] = distance[current] + 1;
				path[w] = current;
			}
		}
	}
}
